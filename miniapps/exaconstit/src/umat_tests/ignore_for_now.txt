#------------------------------------------------------------------------------
# Sources
#------------------------------------------------------------------------------
set(UMAT_HEADERS
    userumat.h
    )

set(UMAT_SOURCES
    umat.f
    userumat.cxx
    )

#------------------------------------------------------------------------------
# Dependencies
#------------------------------------------------------------------------------
set(UMAT_DEPENDS)

umat_fill_depends_list(LIST_NAME  UMAT_DEPENDS
                             DEPENDS_ON  )

include_directories(BEFORE ${PROJECT_BINARY_DIR})

#------------------------------------------------------------------------------
# Defines
#------------------------------------------------------------------------------
set(UMAT_DEFINES HAVE_UMAT)

SET(CMAKE_C_FLAGS_DEBUG   "${CMAKE_C_FLAGS_DEBUG}   -DDEBUG")
SET(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} -DDEBUG")

#------------------------------------------------------------------------------
# Includes
#------------------------------------------------------------------------------
set(UMAT_INCLUDES   ${CMAKE_SOURCE_DIR}/src/interface
                        ${HEADER_INCLUDE_DIR}
                        ${CMAKE_SOURCE_DIR}/src/)
#------------------------------------------------------------------------------
# Build Targets
#------------------------------------------------------------------------------

blt_add_library(NAME    umat_static
                        OUTPUT_NAME umat
                        DEFINES     ${UMAT_DEFINES}
                        INCLUDES    ${UMAT_INCLUDES}
                        DEPENDS_ON  ${UMAT_DEPENDS}
                        HEADERS     ${UMAT_HEADERS}
                        SOURCES     ${UMAT_SOURCES}
                        SHARED      FALSE)
